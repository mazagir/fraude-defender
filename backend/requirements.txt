# ─── AegisShield Backend — Python Dependencies ─────────────────────────────
# Instalar con: pip install -r requirements.txt

# Framework principal
fastapi
uvicorn[standard]

# Base de datos
sqlalchemy
psycopg2-binary          # Driver PostgreSQL para Supabase, Neon y cualquier PostgreSQL en producción

# Autenticación y seguridad
python-jose[cryptography] # JWT (JSON Web Tokens)
passlib[bcrypt]           # Hashing de contraseñas con bcrypt
bcrypt<5                  # passlib 1.7.x no es compatible con bcrypt 5.x
python-multipart          # Soporte para formularios OAuth2 (login form)

# Configuración
python-dotenv             # Cargar variables de entorno desde .env

# Validación de datos
pydantic
pydantic[email]
email-validator

# Rate limiting
slowapi                   # Rate limiting para FastAPI basado en limits/limits

# Cliente HTTP asíncrono (para llamadas a Gemini API)
httpx
pyotp
qrcode[pil]

# Testing
pytest
pytest-asyncio            # Para tests async con FastAPI TestClient
